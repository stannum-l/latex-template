\newcommand{\delimit}[4]{
	\ifcase #1
		\mathopen{}\left#2 #3 \right#4\mathclose{}
		\or #2 #3 #4
		\or \bigl#2 #3 \bigr#4
		\or \Bigl#2 #3 \Bigr#4
		\or \biggl#2 #3 \biggr#4
		\or \Biggl#2 #3 \Biggr#4
	\fi
}

%Generalize builder with a bar
\newcommand{\delb}[5]{
	\delimit{#1}{#2}{
		#3
		\ifcase #1
			\mid
			\or \mid
			\or \bigm|
			\or \Bigm|
			\or \biggm|
			\or \Biggm|
		\fi
		#4
	}{#5}
}

%Generalize builder with a colon
\newcommand{\delc}[5]{
	\delimit{#1}{#2}{
		#3
		\ifcase #1
			:
			\or :
			\else \,:\,
		\fi
		#4
	}{#5}
}
