\begin{Problem}
    Let $G = (V,E)$ be a graph and $M \subseteq E$ be a matching.  Prove
    that $M$ is a maximum matching if and only if there is no augmenting path.
\end{Problem}

    $\ifind$ Prove by contrapositive. Suppose $M$ is not maximum, so there is a matching $M'$ where $\abs{M'} > \abs M$. Let $e = uv$ be an edge where $e \in M' \setminus M$.


    $\iffind$
\end{Proof}

\begin{theorem}
    Let $(f_n)$ be a sequence of continuous functions on $[a,b]$. Suppose
    there is $M$ such that $\abs{f_n(x)} \le M$ for all $x \in [a,b]$ and for all $n$.
    Suppose furthermore that $f(x)$ is continuous function such that
    $\ilim{n} f_n(x) = f(x)$ for ever $x \in [a,b]$. Then:
    \[ \ilim{n} \int_a^b f_n(t) \dif t = \int_a^b f(t) \dif t.\]
\end{theorem}
